3.(hr/root tabla employees) Escribir un procedimiento q visualice el apellido y el salario de los cinco empleados que tienen el salario más alto (Cursor)

SET SERVEROUTPUT ON
CREATE OR REPLACE PROCEDURE ej3(c1 out sys_refcursor) is
begin
	OPEN c1 FOR select last_name ,salary from employees order by salary DESC;
END ej3;
/

BLOQUE ANONIMO:
SET SERVEROUTPUT ON
DECLARE
c sys_refcursor;
ape employees.last_name%type;
sal employees.salary%type;
i number:=0;
BEGIN
ej3(c);
LOOP 
	fetch c into ape,sal;
	exit when c%NOTFOUND or i>=5;
	DBMS_OUTPUT.PUT_LINE('Apellido:' || ape ||' Salario:' || sal);
	i:=i+1;
end loop;
close c;
END;
/
